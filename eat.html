Call, Apply 的区别

call 用法 
  
    var arr = {num: 2};
    var func = function (val,val2) {
      return this.num+val;
    };
    console.log(func.call(arr, 2));  //functionname.call(obj, functionArguments)
apply 用法 
  
    var obj = {num: 2};
    var obj2 = {num: 5};
    var add = function(a,b,c) {
      return this.num+a+b+c;
    }
    var arr = [1,2,3];
    console.log(add.apply(obj, arr)); //functionname.call(obj, functionArguments)
  
  
  
    var obj = {num: 2};

    var add = function(a,b,c) {
      return this.num+a+b+c;

    var obj = {num: 5};
    var add = function(a,b,c) {
      return this.num+a+b+c;
    }
    var bound = add.bind(obj);
